year=year(aa$acct_creation_date)

adate=as.Date(aa$acct_creation_date,units="days")
ldate=as.Date(aa$loan_date,units="days")
adate=as.numeric(adate)
ldate=as.numeric(ldate)
diff=ldate-adate
aa %>%
  ggplot() +
  geom_boxplot(aes(x=loan_default,y=diff),width=0.3,color=c("orange","light green"))+
  labs(title = "Relationship between time and loan condition") +
  xlab("Loan default") +
  ylab("Time Difference between Account creation date and Loan start date")+
  theme(plot.title = element_text (hjust = 0.5))

cat("Bad loans are little more than good loans if the time difference between the start date of loan and account creation date is longer, which means longer time differences will have higher risks to get bad loans.")
